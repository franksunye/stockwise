import { useState, useEffect } from 'react';
import { getWatchlist } from '@/lib/storage';

import { MEMBERSHIP_CONFIG } from '@/lib/membership-config';
import { isPushSupported, subscribeUserToPush } from '@/lib/notifications';

interface Props {
  isOpen: boolean;
  onClose: () => void;
}

export function UserCenterDrawer({ isOpen, onClose }: Props) {
  const isHighPerformance = shouldEnableHighPerformance();
  const [watchlistCount, setWatchlistCount] = useState(0);
  const [userId, setUserId] = useState<string>('');
  const [tier, setTier] = useState<'free' | 'pro'>('free');
  const [expiresAt, setExpiresAt] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);
  
  // Redemption State
  const [redeemCode, setRedeemCode] = useState('');
  const [redeeming, setRedeeming] = useState(false);
  
  // Identity Restore State (for iOS PWA)
  const [showRestoreInput, setShowRestoreInput] = useState(false);
  const [restoreId, setRestoreId] = useState('');
  const [restoring, setRestoring] = useState(false);
  const [restoreMsg, setRestoreMsg] = useState<{type: 'success'|'error', text: string} | null>(null);
  const [redeemMsg, setRedeemMsg] = useState<{type: 'success'|'error', text: string} | null>(null);

  // Notification State
  const [_pushPermission, setPushPermission] = useState<NotificationPermission>('default');
  const [isSubscribing, setIsSubscribing] = useState(false);
  const [pushSupported, setPushSupported] = useState(false);
  const [testingPush, setTestingPush] = useState(false);
  const [isSubscribed, setIsSubscribed] = useState(false);

  // Notification Settings State
  const [showNotificationSettings, setShowNotificationSettings] = useState(false);
  const [notificationSettings, setNotificationSettings] = useState<{
    enabled: boolean;
    types: Record<string, { enabled: boolean; priority: string }>;
  }>({
    enabled: true,
    types: {
      signal_flip: { enabled: true, priority: 'high' },
      morning_call: { enabled: true, priority: 'medium' },
      validation_glory: { enabled: true, priority: 'medium' },
      prediction_updated: { enabled: true, priority: 'low' },
      daily_brief: { enabled: true, priority: 'low' },
      price_update: { enabled: false, priority: 'low' },
    },
  });

  // µ╡ïΦ»òµÄ¿ΘÇüΘÇÜτƒÑ
  const handleTestPush = async () => {
    if (testingPush) return;
    setTestingPush(true);
    setRedeemMsg(null);
    
    try {
      // µúÇµƒÑ Service Worker µÿ»σÉªσ░▒τ╗¬
      const registration = await navigator.serviceWorker.ready;
      if (!registration) {
        setRedeemMsg({ type: 'error', text: 'Service Worker µ£¬σ░▒τ╗¬' });
        return;
      }
      
      // ΘÇÜΦ┐ç Service Worker σÅæΘÇüµ£¼σ£░µ╡ïΦ»òΘÇÜτƒÑ
      await registration.showNotification('≡ƒöö µ╡ïΦ»òΘÇÜτƒÑ - StockWise', {
        body: `µ╡ïΦ»òµêÉσèƒ∩╝üσ╜ôσëìµù╢Θù┤: ${new Date().toLocaleTimeString('zh-CN')}`,
        icon: '/logo.png',
        badge: '/logo.png',
        tag: 'test-notification',
        data: { url: '/dashboard' }
      });
      
      setRedeemMsg({ type: 'success', text: 'µ╡ïΦ»òΘÇÜτƒÑσ╖▓σÅæΘÇü∩╝ü' });
      setTimeout(() => setRedeemMsg(null), 3000);
    } catch (e) {
      console.error('Test push failed:', e);
      setRedeemMsg({ type: 'error', text: 'µ╡ïΦ»òσñ▒Φ┤Ñ: ' + (e instanceof Error ? e.message : String(e)) });
    } finally {
      setTestingPush(false);
    }
  };

  useEffect(() => {
    const initUser = async () => {
      if (!isOpen) return;
      
      // 1. τ╗ƒΣ╕ÇΘÇÜΦ┐ç getCurrentUser ΦÄ╖σÅû/τöƒµêÉτö¿µê╖ ID
      const user = await getCurrentUser();
      setUserId(user.userId);

      // 2. ΦÄ╖σÅûµ£¼σ£░τ¢æµÄºµò░µì«
      const list = getWatchlist();
      setWatchlistCount(list.length);

      // 3. ΦÄ╖σÅûµ£ìσèíτ½»Σ╝Üσæÿτè╢µÇü
      const referredBy = localStorage.getItem('STOCKWISE_REFERRED_BY');
      fetchProfile(user.userId, referredBy);
    };
    initUser();
  }, [isOpen]);

  useEffect(() => {
    const checkPushState = async () => {
        if (typeof window !== 'undefined' && isOpen) {
            setPushSupported(isPushSupported());
            if ('Notification' in window) {
                setPushPermission(Notification.permission);
            }
            // µúÇµƒÑµÿ»σÉªσ╖▓µ£ëΦ«óΘÿà
            if ('serviceWorker' in navigator && Notification.permission === 'granted') {
                try {
                    const registration = await navigator.serviceWorker.ready;
                    const subscription = await registration.pushManager.getSubscription();
                    setIsSubscribed(!!subscription);
                } catch (e) {
                    console.error('Error checking subscription:', e);
                }
            }
        }
        
        // Load notification settings if subscribed
        if (isSubscribed && userId) {
          try {
            const res = await fetch(`/api/user/notification-settings?userId=${userId}`);
            const data = await res.json();
            if (data.settings) {
              setNotificationSettings(data.settings);
            }
          } catch (e) {
            console.error('Failed to load notification settings:', e);
          }
        }
    };

  }, [isOpen, isSubscribed, userId]);

  const handleEnableNotifications = async () => {
    setIsSubscribing(true);
    setRedeemMsg(null);
    
    try {
        // 0. σçåσñçΦ║½Σ╗╜
        let currentUserId = userId;
        if (!currentUserId) {
            const user = await getCurrentUser();

        if (!registration) {
            setRedeemMsg({ type: 'error', text: 'Service Worker µ│¿σåîσñ▒Φ┤Ñ' });

        // 2. Φ»╖µ▒éΘÇÜτƒÑµ¥âΘÖÉ

        if (perm !== 'granted') {
             perm = await Notification.requestPermission();
             setPushPermission(perm);
        }

        if (perm === 'granted') {
            // 3. ΦÄ╖σÅû/σê¢σ╗║Φ«óΘÿà
            const swRegistration = await navigator.serviceWorker.ready;
            let subscription = await swRegistration.pushManager.getSubscription();
            
            if (!subscription) {

            if (subscription) {
                // 4. Σ┐¥σ¡ÿσê░σÉÄτ½»
                const response = await fetch('/api/notifications/subscribe', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ userId: currentUserId, subscription: subscription.toJSON() })
                });
                
                if (response.ok) {
                    setIsSubscribed(true);
                    setRedeemMsg({ type: 'success', text: 'ΘÇÜτƒÑσ╝ÇσÉ»µêÉσèƒ' });
                    setTimeout(() => setRedeemMsg(null), 3000);
                } else {
                    const data = await response.json().catch(() => ({}));
                    setRedeemMsg({ type: 'error', text: 'Σ┐¥σ¡ÿσñ▒Φ┤Ñ: ' + (data.error || response.status) });
                }
            } else {
                 setRedeemMsg({ type: 'error', text: 'µùáµ│òΦÄ╖σÅûµÄ¿ΘÇüµ¥âΘÖÉ' });
            }
        } else {
            setRedeemMsg({ type: 'error', text: 'Φ»╖σàüΦ«╕ΘÇÜτƒÑµ¥âΘÖÉ' });
        }
    } catch (e) {
        console.error('≡ƒöö [Push] Error:', e);
        setRedeemMsg({ type: 'error', text: 'σ╝ÇσÉ»σñ▒Φ┤Ñ∩╝îΦ»╖ΘçìΦ»ò' });
    } finally {
        setIsSubscribing(false);
    }
  };

  // σÅûµ╢êΦ«óΘÿàµÄ¿ΘÇüΘÇÜτƒÑ
  const handleDisableNotifications = async () => {
    setIsSubscribing(true);
    console.log('≡ƒöö [Push] Starting unsubscribe flow...');
    
    try {
        // τí«Σ┐¥ userId σ╖▓σèáΦ╜╜

            const user = await getCurrentUser();
            currentUserId = user.userId;
        }

        const subscription = await swRegistration.pushManager.getSubscription();

            // 3. ΘÇÜτƒÑσÉÄτ½»σêáΘÖñΦ«óΘÿàΦ«░σ╜ò
            const response = await fetch('/api/notifications/unsubscribe', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ userId: currentUserId, endpoint })
            });
            
            if (response.ok) {
                console.log('≡ƒöö [Push] Γ£à Subscription removed from backend');
                setIsSubscribed(false);
                setPushPermission('default'); // Θçìτ╜«µ¥âΘÖÉτè╢µÇüµÿ╛τñ║
                setRedeemMsg({ type: 'success', text: 'σ╖▓σà│Θù¡ΘÇÜτƒÑ' });
                setTimeout(() => setRedeemMsg(null), 3000);
            } else {
                console.error('≡ƒöö [Push] Γ¥î Backend unsubscribe failed');
                setRedeemMsg({ type: 'error', text: 'σÉÄτ½»σÅûµ╢êσñ▒Φ┤Ñ' });
            }
        } else {
            console.log('≡ƒöö [Push] No subscription to remove');
            setIsSubscribed(false);
            setRedeemMsg({ type: 'success', text: 'σ╖▓σà│Θù¡ΘÇÜτƒÑ' });
            setTimeout(() => setRedeemMsg(null), 3000);
        }
    } catch (e) {
        console.error('≡ƒöö [Push] Γ¥î Unsubscribe error:', e);

    } finally {
        setIsSubscribing(false);
    }
  };

  const fetchProfile = async (uid: string, referredBy: string | null = null) => {
      setLoading(true);
      try {
          // ΦÄ╖σÅûµ£¼σ£░ watchlist

              body: JSON.stringify({

                  referredBy: referredBy
              })
          });
          const data = await res.json();
          if (data.tier) {
              setTier(data.tier);
              setExpiresAt(data.expiresAt);
              
              // Σ╜┐τö¿Σ║æτ½»Φ┐öσ¢₧τÜä watchlistCount (source of truth)

              }
              
              // σÉîµ¡ÑµêÉσèƒσÉÄ∩╝îσªéµ₧£Φ«░σ╜òΣ║å invite∩╝îσÅ»Σ╗Ñµ╕àΘÖñΣ╗ÑΘÿ▓ΘçìσñìµÅÉΣ║ñ∩╝êσÅ»ΘÇë∩╝ë

          }

            console.error(e);
        } finally {

        }

              setTimeout(() => setRedeemMsg(null), 3000);
          } else {
              setRedeemMsg({ type: 'error', text: data.error || 'µ┐Çµ┤╗σñ▒Φ┤Ñ' });
          }
        } catch {
            setRedeemMsg({ type: 'error', text: 'τ╜æτ╗£Φ»╖µ▒éσñ▒Φ┤Ñ' });
        } finally {
            setRedeeming(false);
        }

            animate={{ y: 0 }}
            exit={{ y: '100%' }}
            drag="y"
            dragConstraints={{ top: 0, bottom: 0 }}
            dragElastic={{ top: 0.1, bottom: 0.6 }}
            onDragEnd={(_, info) => {
              if (info.offset.y > 150) onClose();
            }}
            transition={isHighPerformance 
              ? { type: 'tween', ease: 'easeOut', duration: 0.25 }
              : { type: 'spring', damping: 25, stiffness: 200 }
            }
            className="w-full max-w-md bg-[#0a0a0f] border-t border-white/10 rounded-t-[32px] shadow-[0_-20px_50px_rgba(0,0,0,0.5)] overflow-hidden pointer-events-auto"
          >
            {/* ΦºåΦºëµïëµëï */}
            <div className="w-full flex justify-center pt-3 pb-1">
               <div className="w-12 h-1 rounded-full bg-white/20" />
            </div>

            <div className="p-8 pt-4 flex flex-col min-h-[60vh]">
              <header className="flex items-center justify-between mb-8">
                <div className="space-y-1.5">
                   <div className="flex items-center gap-2">
                    <div className="w-1.5 h-1.5 rounded-full bg-indigo-500 shadow-[0_0_8px_#6366f1] animate-pulse" />
                    <span className="text-[10px] uppercase tracking-[0.4em] text-slate-500 font-bold">Member Center</span>
                   </div>
                   <h2 className="text-3xl font-black italic tracking-tighter text-white">
                    Σ╕¬Σ║║Σ╕¡σ┐â <span className="text-indigo-500 underline decoration-4 underline-offset-4" data-en="USER">USER</span>

              {/* τö¿µê╖σìíτëç */}

                   <div className={`w-16 h-16 rounded-full border-2 flex items-center justify-center relative ${tier === 'pro' ? 'border-amber-500/50 bg-amber-500/10' : 'bg-white/5 border-white/10'}`}>
                      {loading ? <Loader2 className="w-8 h-8 text-slate-400 animate-spin" /> : <User className={`w-8 h-8 ${tier === 'pro' ? 'text-amber-200' : 'text-slate-400'}`} />}
                   </div>

                           tier === 'pro' 
                           ? 'bg-amber-500/20 border-amber-500/30 text-amber-400' 
                           : 'bg-slate-700/50 border-white/5 text-slate-300'
                       }`}>
                           {tier === 'pro' ? 'Pro Plan' : 'Free Plan'}
                       </span>
                     </div>
                      <p className="text-xs text-slate-500 mono flex items-center gap-2">
                        ID: {userId.slice(0, 12)}...

                          onClick={() => setShowRestoreInput(!showRestoreInput)}
                          className="text-slate-600 hover:text-indigo-400 transition-colors p-0.5"
                          title="µüóσñìΦ║½Σ╗╜"
                        >
                          <Key size={10} />
                        </button>
                      </p>
                     {expiresAt && tier === 'pro' && (

                             µ£ëµòêµ£ƒΦç│: {expiresAt.split('T')[0]}
                         </p>
                     )}
                   </div>
                </div>
              </div>

              {/* Φ╡äµ║Éτö¿ΘçÅ */}
              <div className="grid grid-cols-2 gap-3 mb-4">
                <div className="glass-card p-5 flex flex-col justify-between h-24">
                   <div className="flex items-center gap-2 text-slate-400">
                     <ShieldCheck size={16} />
                     <span className="text-xs font-bold uppercase">τ¢æµÄºΘàìΘó¥</span>

                     <span className="text-sm font-bold text-slate-600 mb-1">/ {tier === 'pro' ? '10' : '3'}</span>
                   </div>

                   <div className={`flex items-center gap-2 ${tier === 'pro' ? 'text-amber-200' : 'text-slate-400'}`}>

                     </span>

                </div>
              </div>

              {/* Pricing Page Link */}
              <div className="mb-6 px-1">
                 <Link 
                   href="/pricing"
                   className="w-full py-4 px-5 rounded-[24px] bg-gradient-to-r from-indigo-500/5 to-purple-500/5 border border-white/5 hover:border-indigo-500/20 transition-all flex items-center justify-between group"
                   onClick={() => onClose()}
                 >
                    <div className="flex items-center gap-3">

                            <Star size={18} className="text-indigo-400" />
                        </div>
                        <div>
                            <span className="block text-sm font-black italic text-white">µƒÑτ£ïΦ»ªτ╗åΣ╗╖µá╝µ¥âτ¢è</span>

                    <ChevronRight size={16} className="text-slate-600 group-hover:text-indigo-400 transition-colors" />

              </div>

              {/* Notification Switch (PWA Only) - Upgraded with Type Settings */}
              {pushSupported && (

                                    <p className="text-[10px] text-slate-500">AI Σ┐íσÅ╖Σ╕ÄµùÑµèÑµÅÉΘåÆ</p>
